Which implementation is better?

```javascript
// Option 1: reduce
const sum = numbers.reduce((total, num) => total + num, 0);
const max = Math.max(...numbers);
const min = Math.min(...numbers);
return { min, max, sum };
```

```javascript
// Option 2: loop
let sum = 0;
let min = numbers[0];
let max = numbers[0];
for (let i = 0; i < numbers.length; i++) {
  sum += numbers[i];
  if (min > numbers[i]) {
    min = numbers[i];
  }

  if (max < numbers[i]) {
    max = numbers[i];
  }
}

return { min, max, sum };
```
